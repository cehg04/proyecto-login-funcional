create table permisos (
cod_usuario int not null,
cod_opcion int not null,
permiso char(1) not null default 'N' comment 'S = si , N = no',
primary key (cod_usuario, cod_opcion),
foreign key (cod_usuario) references usuarios(cod_usuario),
foreign key (cod_opcion) references opciones(cod_opcion)
);

create table empresas (
cod_empresa int not null primary key,
nombre varchar(100) null,
abreviatura varchar(50) not null,
estado char(1) default 'A' null comment 'A = activo , I = inactivo'
);

CREATE TABLE proveedores (
cod_proveedor VARCHAR(20) NOT NULL,
cod_empresa INT NOT NULL,
nombre VARCHAR(100) NOT NULL,
nit VARCHAR(20) NOT NULL,
estado CHAR(1) DEFAULT 'A' NULL COMMENT 'A = activo , I = inactivo',
PRIMARY KEY (cod_proveedor, cod_empresa),
FOREIGN KEY (cod_empresa) REFERENCES empresas(cod_empresa)
);

INSERT INTO monedas (cod_moneda, abreviatura) VALUES 
('USD', '$'),
('GTQ', 'Q');

create table monedas (
cod_moneda char(3) not null primary key,
abreviatura varchar(1) not null 
);


create table tipo_documentos (
cod_tipo_documento int not null primary key,
nombre_documento varchar(50) not null,
estado char(1) not null comment 'A = activo , I = inactivo'
);

CREATE TABLE enca_contrasenias (
cod_contrasenia INT NOT NULL,
cod_empresa INT NOT NULL,
cod_empresa_proveedor INT NOT NULL,
num_contrasenia VARCHAR(20) NOT NULL,
cod_proveedor VARCHAR(20) NOT NULL,
fecha_contrasenia DATE NOT NULL,
usuario_creacion INT NOT NULL,
fecha_creacion DATETIME NOT NULL,
estado CHAR(1) DEFAULT 'R' NOT NULL COMMENT 'R = Realizado / X = Anulado',
usuario_x INT NULL,
fecha_x DATETIME NULL,
comentario VARCHAR(255) NULL COMMENT 'Motivo o detalle de la anulación',
PRIMARY KEY (cod_contrasenia, cod_empresa),
FOREIGN KEY (cod_empresa) REFERENCES empresas(cod_empresa),
FOREIGN KEY (usuario_creacion) REFERENCES usuarios(cod_usuario),
FOREIGN KEY (usuario_x) REFERENCES usuarios(cod_usuario),
FOREIGN KEY (cod_proveedor, cod_empresa_proveedor) 
REFERENCES proveedores(cod_proveedor, cod_empresa)
);

create table detalle_contrasenias (
cod_contrasenia int not null,
cod_empresa int not null,
linea int not null,
num_factura int not null,
cod_moneda char(3) not null,
monto decimal(19,4) not null,
retension_iva char(1) null comment 'S = si , N = no',
retension_isr char(1) null comment 'S = si , N = no',
numero_retension_iva int null,
numero_retension_isr int null,
estado char(1) default 'P' not null comment 'P = pendiente, E = entregado, R = recibido',
primary key (cod_contrasenia, cod_empresa, linea),
foreign key (cod_contrasenia, cod_empresa) 
	references enca_contrasenias(cod_contrasenia, cod_empresa),
foreign key (cod_moneda) references monedas(cod_moneda)
);

drop table documentos_varios;
create table documentos_varios (
cod_documento int not null,
cod_empresa int not null,
cod_tipo_documento int not null,
cod_proveedor varchar(20)  null,
nombre_solicitud varchar(100) null,
numero_documento int null,
cod_moneda char(3) not null,
monto decimal(19,4) not null,
observaciones varchar(255) null,
estado char(1) not null comment 'P = pendiente, E = entregado, R = recibido, X = anulado',
primary key (cod_documento),
foreign key (cod_empresa) references empresas(cod_empresa),
foreign key (cod_tipo_documento) references tipo_documentos(cod_tipo_documento),
foreign key (cod_proveedor) references proveedores(cod_proveedor),
foreign key (cod_moneda) references monedas(cod_moneda)
);

drop table enca_entregas;
create table enca_entregas (
cod_entrega int not null,
cod_empresa int not null,
num_entrega varchar(20) not null,
fecha_entrega date not null,
usuario_creacion int not null,
fecha_creacion datetime not null,
estado char(1) default 'P' not null comment 'P = pendiente ,R= recibido , X = anulado',
tipo_entrega char(2) not null comment 'DC = documento con contraseña , DS = documento sin contraseña',
usuario_x int  null,
fecha_x datetime  null,
primary key (cod_entrega, cod_empresa),
foreign key (cod_empresa) references empresas(cod_empresa),
foreign key (usuario_creacion) references usuarios(cod_usuario),
foreign key (usuario_x) references usuarios(cod_usuario)
);

drop table detalle_entregas;
create table detalle_entregas (
cod_entrega int not null,
cod_empresa int not null,
linea int not null,
cod_contrasenia int null,
cod_empresa_contrasenia int null,
linea_contrasenia int null,
num_factura int null ,
cod_moneda char(3) not null,
monto decimal(19,4) not null,
retension_iva char(1) null comment 'S = si , N = no',
retension_isr char(1) null comment 'S = si , N = no',
numero_retension_iva int null,
numero_retension_isr int null,
estado char(1) default 'P' not null comment 'p = pendiente, C = confirmado',
cod_documento int null ,
primary key (cod_entrega, cod_empresa, linea),
foreign key (cod_entrega, cod_empresa)
	references enca_entregas(cod_entrega, cod_empresa),
foreign key (cod_contrasenia, cod_empresa_contrasenia, linea_contrasenia)
  references detalle_contrasenias(cod_contrasenia, cod_empresa, linea),
foreign key (cod_documento) references documentos_varios(cod_documento),
foreign key (cod_moneda) references monedas(cod_moneda)
);
